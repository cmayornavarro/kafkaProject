


#create topic with partitions
bin\windows\kafka-topics.bat --create --topic christianTopic --replication-factor 1 --partitions 2 --bootstrap-server localhost:9092

#list topics
bin\windows\kafka-topics.bat --list --bootstrap-server localhost:9092
>christianTopic

#add a consumer1 with group groupRCU
bin\windows\kafka-console-consumer.bat --topic christianTopic --group groupRCU --bootstrap-server localhost:9092 

#list consumers groups
bin\windows\kafka-consumer-groups.bat --list --bootstrap-server localhost:9092

>groupRCU
